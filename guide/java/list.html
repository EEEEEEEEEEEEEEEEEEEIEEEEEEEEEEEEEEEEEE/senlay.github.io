  • List中循环取出最小、最大值
 String temp = null ;
 List<String> strList = new ArrayList<>();
   strList.add("6"); 
   strList.add("8");
   strList.add("10");
 for(String str : strList){
     if(StringUtils.isEmpty(temp)){
         temp = str;
     }
     if(StringUtils.isNotEmpty(str)){
         temp = compareTos(temp,str);
     }
 }
 System.out.print("最大值："+temp)

public static String compareTos(String str1, String str2){
    if(str1.compareTo(str2) > 0){
        return str1;
    }
    return str2;
}
